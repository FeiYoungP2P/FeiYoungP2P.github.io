java 界面组件显示不及时

很头疼，用JFrame做界面时总是遇到各种组件显示不及时，经过各种蛇皮操作，终于找到了解决方法！

我的部分界面代码如下：

记住一定要把JFrame的操作代码放在组件添加代码的后面！

public abstract class Gui implements ActionListener {

​ 

public JFrame f = new JFrame("相似单词查询");
public JPanel jp = new JPanel();
public JButton jb = new JButton("查询");
public JTextField jtf = new JTextField();
public JTextArea jta1 = new JTextArea();
public JTextArea jta2 = new JTextArea();
public Gui() {
    // TODO Auto-generated constructor stub

    jp.setLayout(null);

    JLabel lab1 = new JLabel("输入单词：");
    lab1.setBounds(30, 20, 80, 30);
    jp.add(lab1);

​
jtf.setBounds(110, 20, 300, 30);
jtf.setFont(new Font(“Timesnewroman”, Font.PLAIN, 18));
jp.add(jtf);

​
jb.setBounds(460, 20, 60, 30);
jb.setMargin(new Insets(0,0,0,0));
jb.addActionListener(this);
jp.add(jb);

JLabel lab2 = new JLabel("查询结果：");
lab2.setBounds(30, 100, 80, 30);
jp.add(lab2);

​
jta1.setEditable(true);
jta1.setLineWrap(true);
jta1.setFont(new Font(“Timesnewroman”, Font.PLAIN, 18));
jta1.setBounds(110, 110, 170, 300);
JScrollPane jsp = new JScrollPane(jta1);
jsp.setBounds(110, 110, 170, 300);
jp.add(jsp);

jta2.setEditable(true);
jta2.setLineWrap(true);
jta2.setFont(new Font("Timesnewroman", Font.PLAIN, 18));
jta2.setBounds(300, 110, 170, 300);
JScrollPane jsp2 = new JScrollPane(jta2);
jsp2.setBounds(300, 110, 170, 300);
jp.add(jsp2);

​
f.setBounds(550, 300, 600, 500);
f.setResizable(false);
f.setVisible(true);
f.setContentPane(jp);
f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
}
}
